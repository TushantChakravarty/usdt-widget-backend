/**
 * Main entry point for the /api/v1 route.
 * @file Index Routes
 * @module Index Routes
 * @category routes
 */

import userRoute from "./user.route.js";
import adminRoute from "./admin.route.js";
import mediaRoute from "./media.route.js";
import socialRoute from "./social.route.js"
import depositRoute from "./deposit.route.js";
import transactionRoute from "./transaction.route.js"
import offerRoute from "./offer.route.js"
import withdrawRoute from "./withdraw.route.js";
import page from "./page.route.js"

const routes = async (route, options) => { // route = fastify instance
  // Registers user routes under the prefix '/user' (e.g., /api/v1/user)
  route.register(userRoute, { prefix: "/user" })

  // Registers admin routes under the prefix '/admin' (e.g., /api/v1/admin)
  route.register(adminRoute, { prefix: "/admin" })

  // Registers media routes under the prefix '/media' (e.g., /api/v1/media)
  route.register(mediaRoute, { prefix: "/media" })

  // Registers social routes under the prefix '/social' (e.g., /api/v1/social)
  route.register(socialRoute, { prefix: "/social" })

  //Registers payment routes under the prefix '/payment' (e.g, /api/v1/deposit)
  route.register(depositRoute, { prefix: "/deposit" })

  // Registers transaction routes under the prefix '/transaction' (e.g., /api/v1/transaction)
  route.register(transactionRoute, { prefix: "/transaction" })

  //Registers offer routes under the prefix '/offer' (e.g.,/api/v1/offer)
  route.register(offerRoute, { prefix: "/offer" })

  // Registers withdraw routes under the prefix '/withdraw' (e.g., /api/v1/withdraw)
  route.register(withdrawRoute, { prefix: "/withdraw" })

  // Registers Page routes under the prefix '/page' (e.g., /api/v1/page)
  route.register(page, { prefix: "/page" })
};

export default routes;
