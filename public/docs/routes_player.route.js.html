/**
 * Router for player-related endpoints.
 * Handles player authentication, user management, and access control operations.
 * @file Admin Routes
 * @module Player Routes
 * @category routes
 * @subcategory player
 */

import Controller from "../controllers"
import Validator from "../validators";

const { Player: player } = Controller;
const { Player: validator } = Validator;

const routes = async (route, options) => { // route = fastify instance

    /**
     * Route for getting all player playing on the particular match.
     * Requires authentication and admin,user role.
     * Request parameters should include the match ID.
     */
    route.get("/:id", {
        onRequest: [
            route.authenticate, // 1.authentication middleware
            async function (request, reply) {
                request.accessRoles = ["admin", "user"]; // 2.allowed role for this route
                request.accessFields = ["admin"]; // 3.allowed access fields for this route
            },
            route.rbam, // 4.RBAM middleware
           // route.redisMiddleware // 5.Redis cache middleware

        ],
        schema: validator.getPlayersByMatchId,
        handler: player.getPlayersByMatchId,
    });

};

export default routes;
