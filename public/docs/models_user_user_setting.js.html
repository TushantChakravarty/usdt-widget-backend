/**
 * User setting definition.
 * Represents user's settings/preferences in the database.
 * @module models/user_setting
 * @requires sequelize
 * @requires sequelize/Model
 * @exports UserSetting
 * @class UserSetting
 */

"use strict";
import { Model } from "sequelize";
// Setting model.
export default (sequelize, DataTypes) => {
    class UserSetting extends Model {
        static associate(models) {
            // associations
            UserSetting.belongsTo(models.User, {
                foreignKey: "user_id",
                as: "user",
            });
        }
    }
    UserSetting.init(
        {

            user_id: { type: DataTypes.INTEGER, unique: true, allowNull: false },
            allow_whatsapp_messages: { // this field is just to access play999 updates,offers and support
                type: DataTypes.BOOLEAN,
                defaultValue: true
            },
            display_full_name: { // this field is just to show users fullname to other user
                type: DataTypes.BOOLEAN,
                defaultValue: false
            },
            show_my_previous_teams: { // this field is just to show your previous teams to other users
                type: DataTypes.BOOLEAN,
                defaultValue: true
            },
            show_which_contest_i_join: { // this field is just to show what kind of contest you have joined for completed matches
                type: DataTypes.BOOLEAN,
                defaultValue: true
            },
            allow_challenges: {
                type: DataTypes.BOOLEAN, // other people can challenge you who are not in your contact
                defaultValue: true
            },

            // TODO :Safety field where user can see blocked person

        },
        {
            sequelize,
            modelName: "UserSetting",
            indexes: [{ unique: true, fields: ["user_id"] }],
        }
    );
    return UserSetting;
};
